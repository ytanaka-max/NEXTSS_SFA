// CSVエクスポート機能（修正版） function exportCSV() { const table = document.getElementById("sfa-table"); let csvContent = ""; const rows = table.querySelectorAll("tr"); for (let i = 0; i < rows.length; i++) { const row = [], cols = rows[i].querySelectorAll("td, th"); for (let j = 0; j < cols.length; j++) { // セル内のテキストを取得し、カンマや改行をエスケープ let data = cols[j].innerText.replace(/"/g, '""'); row.push('"' + data + '"'); } csvContent += row.join(",") + "\r\n"; } // 日本語文字化け防止のためのBOM付与 const bom = new Uint8Array([0xEF, 0xBB, 0xBF]); const blob = new Blob([bom, csvContent], { type: "text/csv;charset=utf-8;" }); // ダウンロード用リンクの生成と実行 if (navigator.msSaveBlob) { // IE用 navigator.msSaveBlob(blob, "NEXTsalesAI_SFA_List.csv"); } else { const link = document.createElement("a"); const url = URL.createObjectURL(blob); link.setAttribute("href", url); link.setAttribute("download", "NEXTsalesAI_SFA_List.csv"); link.style.visibility = 'hidden'; document.body.appendChild(link); link.click(); document.body.removeChild(link); } } // CSVインポート機能（ファイル選択時に自動実行） document.getElementById('csv-upload').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(event) { const text = event.target.result; const lines = text.split(/\r\n|\n/); const tbody = document.querySelector('#sfa-table tbody'); tbody.innerHTML = ''; lines.forEach((line, index) => { if (index === 0 || line.trim() === '') return; const cols = line.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/); const row = document.createElement('tr'); cols.forEach((col, i) => { const td = document.createElement('td'); let val = col.replace(/^"|"$/g, '').replace(/""/g, '"'); if (i === 4 && val === '〇') { td.innerHTML = '〇'; } else if (i === 1) { td.style.color = 'var(--next-blue)'; td.style.fontWeight = '600'; td.innerText = val; } else { td.innerText = val; } row.appendChild(td); }); tbody.appendChild(row); }); alert('インポートが完了しました。100行のデータを表示中。'); }; reader.readAsText(file, "UTF-8"); });
